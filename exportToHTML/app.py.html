259      def _detect_faces(self, frame):
260          gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
261  
262          # Adjusted parameters for better face detection with less false positives
263          faces = self.face_detector.detectMultiScale(
264              gray,
265              scaleFactor=1.1,
266              minNeighbors=6,  # Increased from 5 to 6 for better confidence
267              minSize=(60, 60),  # Set minimum face size
268              flags=cv2.CASCADE_SCALE_IMAGE
269          )
270  
271          face_locations = [(x, y, x + w, y + h) for (x, y, w, h) in faces]
272          return face_locations

274      def _get_face_embedding(self, face_img):
275          face_img = cv2.resize(face_img, (96, 96))
276  
277          face_blob = cv2.dnn.blobFromImage(
278              face_img, 1.0/255, (96, 96), (0, 0, 0), swapRB=True, crop=False
279          )
280  
281          self.face_recognition_model.setInput(face_blob)
282          embedding = self.face_recognition_model.forward()
283  
284          embedding = embedding.flatten()
285          embedding = embedding / np.linalg.norm(embedding)  # Normalize
286  
287          return embedding

289      def _compute_face_id(self, face_location):
290          x1, y1, x2, y2 = face_location
291          face_center = ((x1 + x2) // 2, (y1 + y2) // 2)
292          face_size = (x2 - x1) * (y2 - y1)
293          return f"{face_center[0]}_{face_center[1]}_{face_size}"

295      def _find_nearest_tracked_face(self, face_location, max_distance=80):  # Reduced max distance
296          curr_center_x = (face_location[0] + face_location[2]) // 2
297          curr_center_y = (face_location[1] + face_location[3]) // 2
298  
299          min_distance = float('inf')
300          nearest_id = None
301  
302          for face_id, face_data in self.face_tracking.items():
303              tracked_center_x = (face_data['location'][0] + face_data['location'][2]) // 2
304              tracked_center_y = (face_data['location'][1] + face_data['location'][3]) // 2
305  
306              distance = np.sqrt((curr_center_x - tracked_center_x)**2 + (curr_center_y - tracked_center_y)**2)
307  
308              if distance < min_distance and distance < max_distance:
309                  min_distance = distance
310                  nearest_id = face_id
311  
312          return nearest_id, min_distance

314      def recognize_face(self, face_embedding):
315          if len(self.known_face_encodings) == 0:
316              return None, None, 0.0
317  
318          # Calculate similarities with all known faces
319          similarities = [cosine_similarity([face_embedding], [known_encoding])[0][0]
320                         for known_encoding in self.known_face_encodings]
321  
322          best_match_index = np.argmax(similarities)
323          best_similarity = similarities[best_match_index]
324  
325          if best_similarity >= self.recognition_threshold:
326              person_id = self.known_face_names[best_match_index]
327              name = person_id
328              if person_id.startswith("EMPLOYEE_"):
329                  name = person_id.replace("EMPLOYEE_", "")
330              elif person_id.startswith("VISITOR_"):
331                  name = f"Visitor {person_id.split('_')[1]}"
332  
333              # Only return positive recognition if similarity is above confidence threshold
334              if best_similarity >= self.recognition_confidence:
335                  return person_id, name, best_similarity
336              else:
337                  # For similarity between threshold and confidence, mark as potential match
338                  return f"POTENTIAL_{person_id}", name, best_similarity
339  
340          return None, None, 0.0

342      def process_new_face(self, frame, face_coords):
343          x1, y1, x2, y2 = face_coords
344          face_image = frame[y1:y2, x1:x2]
345  
346          # Generate a position-based identifier for the face
347          face_position_id = f"{(x1+x2)//2}_{(y1+y2)//2}_{x2-x1}"
348  
349          # Check if we've recently processed a face at this position
350          current_time = time.time()
351          if face_position_id in self.face_detection_history:
352              time_since_last_detection = current_time - self.face_detection_history[face_position_id]['time']
353              if time_since_last_detection < self.face_recapture_timeout:
354                  # Return cached identity if we've seen this face recently
355                  cached_data = self.face_detection_history[face_position_id]
356                  return cached_data['person_id'], cached_data['name'], face_coords, cached_data['similarity']
357  
358          face_id, distance = self._find_nearest_tracked_face(face_coords)
359  
360          # If we have a nearby face that was recently seen, use its identity
361          if face_id and distance < 60:  # Reduced distance threshold
362              if (current_time - self.face_tracking[face_id]['last_seen']) < self.face_recapture_timeout:
363                  self.face_tracking[face_id]['location'] = face_coords
364                  self.face_tracking[face_id]['last_seen'] = current_time
365  
366                  # Update detection history
367                  self.face_detection_history[face_position_id] = {
368                      'time': current_time,
369                      'person_id': self.face_tracking[face_id]['person_id'],
370                      'name': self.face_tracking[face_id]['name'],
371                      'similarity': self.face_tracking[face_id]['similarity']
372                  }
373  
374                  return self.face_tracking[face_id]['person_id'], self.face_tracking[face_id]['name'], face_coords, self.face_tracking[face_id]['similarity']
375  
376          try:
377              face_embedding = self._get_face_embedding(face_image)
378              person_id, name, similarity = self.recognize_face(face_embedding)
379  
380              if person_id is None or person_id.startswith("POTENTIAL_"):
381                  # New visitor or uncertain match
382                  if person_id is None:
383                      # Definitely new visitor
384                      person_id = f"VISITOR_{self.visitor_counter}"
385                      self.visitor_counter += 1
386                      name = person_id
387                      action = "NEW_VISITOR"
388                  else:
389                      # Potential match - use the potential ID but mark as unconfirmed
390                      person_id = person_id.replace("POTENTIAL_", "")
391                      action = "POTENTIAL_MATCH"
392  
393                  # Add to known faces if it's a new visitor
394                  if action == "NEW_VISITOR":
395                      self.known_face_encodings.append(face_embedding)
396                      self.known_face_names.append(person_id)
397                      self.known_face_metadata.append({
398                          'type': 'visitor',
399                          'source': 'live_capture',
400                          'first_seen': datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
401                          'last_seen': datetime.now().strftime("%Y-%m-%d %H:%M:%S")
402                      })
403                      self._save_embeddings()
404  
405                      visitor_dir = os.path.join(self.dirs['visitors'], person_id)
406                      os.makedirs(visitor_dir, exist_ok=True)
407  
408                      timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
409                      face_file = os.path.join(visitor_dir, f"{timestamp}.jpg")
410                      cv2.imwrite(face_file, face_image)
411  
412                  self._log_attendance(person_id, name, action, similarity)
413                  print(f"{action}: {person_id} (Similarity: {similarity:.4f})")
414              else:
415                  # Confirmed recognition
416                  self._log_attendance(person_id, name, "RECOGNIZED", similarity)
417                  print(f"Recognized: {person_id} (Similarity: {similarity:.4f})")
418  
419                  # Update visitor directory if this is a visitor
420                  if person_id.startswith("VISITOR_"):
421                      visitor_dir = os.path.join(self.dirs['visitors'], person_id)
422                      os.makedirs(visitor_dir, exist_ok=True)
423  
424                      timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
425                      face_file = os.path.join(visitor_dir, f"{timestamp}.jpg")
426                      cv2.imwrite(face_file, face_image)
427  
428              new_face_id = f"face_{len(self.face_tracking) + 1}"
429              self.face_tracking[new_face_id] = {
430                  'person_id': person_id,
431                  'name': name,
432                  'location': face_coords,
433                  'last_seen': current_time,
434                  'embedding': face_embedding,
435                  'similarity': similarity
436              }
437  
438              # Update detection history
439              self.face_detection_history[face_position_id] = {
440                  'time': current_time,
441                  'person_id': person_id,
442                  'name': name,
443                  'similarity': similarity
444              }
445  
446              return person_id, name, face_coords, similarity
447  
448          except Exception as e:
449              print(f"Error processing face: {e}")
450              return None, "Error", face_coords, 0.0

471      def _cleanup_stale_tracking(self, max_age=10):
472          current_time = time.time()
473          stale_ids = []
474  
475          for face_id, face_data in self.face_tracking.items():
476              if current_time - face_data['last_seen'] > max_age:
477                  stale_ids.append(face_id)
478  
479          for face_id in stale_ids:
480              del self.face_tracking[face_id]
481  
482          return len(stale_ids)

484      def _frame_processor(self):
485          """Separate thread for processing frames to improve performance"""
486          while self.running:
487              if not self.frame_queue.empty() and self.processing_enabled:
488                  frame, face_locations = self.frame_queue.get()
489  
490                  recognized_info = []
491                  current_time = time.time()
492  
493                  # Process only distinct faces (avoid multiple detections of the same face)
494                  processed_positions = set()
495  
496                  for face_location in face_locations:
497                      # Create a position identifier for the face
498                      x1, y1, x2, y2 = face_location
499                      center_x, center_y = (x1+x2)//2, (y1+y2)//2
500                      face_pos_id = f"{center_x//20}_{center_y//20}"  # Group similar positions
501  
502                      # Skip if we already processed a face in this position
503                      if face_pos_id in processed_positions:
504                          continue
505  
506                      processed_positions.add(face_pos_id)
507  
508                      face_id, distance = self._find_nearest_tracked_face(face_location)
509  
510                      if face_id and distance < 60:  # Reduced distance threshold
511                          if (current_time - self.face_tracking[face_id]['last_seen']) < 1:
512                              self.face_tracking[face_id]['location'] = face_location
513                              self.face_tracking[face_id]['last_seen'] = current_time
514                              recognized_info.append((
515                                  self.face_tracking[face_id]['person_id'],
516                                  self.face_tracking[face_id]['name'],
517                                  face_location,
518                                  self.face_tracking[face_id]['similarity']
519                              ))
520                              continue
521  
522                      if self.auto_capture:
523                          result = self.process_new_face(frame, face_location)
524                          if result:
525                              person_id, name, bbox, similarity = result
526                              recognized_info.append((person_id, name, bbox, similarity))
527  
528                  # Update recognized faces for display
529                  self.recognized_faces = recognized_info

531      def calculate_fps(self):
532          """Calculate and return current FPS"""
533          self.frame_count += 1
534          current_time = time.time()
535          elapsed = current_time - self.last_fps_update
536  
537          if elapsed >= 1.0:  # Update FPS every second
538              self.fps = self.frame_count / elapsed
539              self.frame_count = 0
540              self.last_fps_update = current_time
541  
542          return self.fps

544      def draw_info_on_frame(self, frame, face_locations):
545          display_frame = frame.copy()
546          fps = self.calculate_fps()
547  
548          for face_location in face_locations:
549              x1, y1, x2, y2 = face_location
550              label = "Unknown"
551              color = (0, 165, 255)  # Orange for unknown
552  
553              # Check if this face was recognized in the current processing
554              for person_id, name, bbox, similarity in self.recognized_faces:
555                  if bbox == face_location:
556                      if person_id.startswith("EMPLOYEE_"):
557                          color = (0, 255, 0)  # Green for employees
558                          label = name.replace("EMPLOYEE_", "")
559                      elif person_id.startswith("VISITOR_"):
560                          color = (0, 255, 255)  # Yellow for visitors
561                          label = f"Visitor {person_id.split('_')[1]}"
562                      elif person_id.startswith("POTENTIAL_"):
563                          color = (255, 255, 0)  # Light blue for potential matches
564                          label = f"Maybe {name.replace('EMPLOYEE_', '')}"
565                      break
566  
567              cv2.rectangle(display_frame, (x1, y1), (x2, y2), color, 2)
568              cv2.rectangle(display_frame, (x1, y2-35), (x2, y2), color, cv2.FILLED)
569              cv2.putText(display_frame, label, (x1+6, y2-6), cv2.FONT_HERSHEY_DUPLEX, 0.6, (0, 0, 0), 1)
570  
571          # Status information
572          status_texts = [
573              f"FPS: {fps:.1f}",
574              f"Faces: {len(face_locations)}",
575              f"Auto-capture: {'ON' if self.auto_capture else 'OFF'}",
576              f"Session: {self.session_id}",
577              f"Known faces: {len(self.known_face_names)}"
578          ]
579  
580          for i, text in enumerate(status_texts):
581              cv2.putText(display_frame, text, (10, 30 + i*30),
582                          cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
583  
584          return display_frame

586      def save_frame(self, frame, prefix="frame"):
587          timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
588          frame_path = os.path.join(self.dirs['frames'], f"{prefix}_{timestamp}.jpg")
589          cv2.imwrite(frame_path, frame)
590          return frame_path

592      def run(self):
593          print("\nFace Attendance System")
594          print("Running indefinitely. Press Ctrl+C to stop.")
595  
596          start_time = time.time()
597          save_frame_interval = 10
598          last_frame_save = 0
599          last_cleanup_time = time.time()
600          cleanup_interval = 5
601          last_status_update = 0  # Track last status update time
602  
603          try:
604              while self.running:
605                  ret, frame = self.cap.read()
606                  if not ret:
607                      print("Error capturing frame")
608                      break
609  
610                  frame = cv2.flip(frame, 1)
611                  face_locations = self._detect_faces(frame)
612  
613                  # Clean up stale tracking periodically
614                  current_time = time.time()
615                  if (current_time - last_cleanup_time) > cleanup_interval:
616                      removed = self._cleanup_stale_tracking()
617                      if removed > 0:
618                          print(f"Removed {removed} stale face tracking entries")
619                      last_cleanup_time = current_time
620  
621                  # Add frame to processing queue if not full
622                  if self.frame_queue.qsize() < 2:
623                      self.frame_queue.put((frame.copy(), face_locations))
624  
625                  # Save frame periodically if faces detected
626                  if (current_time - last_frame_save) > save_frame_interval:
627                      if len(face_locations) > 0:
628                          annotated_frame = self.draw_info_on_frame(frame, face_locations)
629                          saved_path = self.save_frame(annotated_frame)
630                          print(f"Saved frame with {len(face_locations)} faces to {saved_path}")
631                          last_frame_save = current_time
632  
633                  display_frame = self.draw_info_on_frame(frame, face_locations)
634  
635                  if not self.headless:
636                      cv2.imshow("Face Attendance System", display_frame)
637  
638                      key = cv2.waitKey(1) & 0xFF
639                      if key == ord('q'):
640                          break
641                      elif key == ord('a'):
642                          self.auto_capture = not self.auto_capture
643                          print(f"Auto-capture {'enabled' if self.auto_capture else 'disabled'}")
644                      elif key == ord('p'):
645                          self.processing_enabled = not self.processing_enabled
646                          print(f"Face processing {'enabled' if self.processing_enabled else 'disabled'}")
647  
648                  # Print status every N seconds (increased interval)
649                  elapsed_seconds = int(time.time() - start_time)
650                  if elapsed_seconds > 0 and (current_time - last_status_update) >= self.status_update_interval:
651                      elapsed_minutes = elapsed_seconds // 60
652                      remaining_seconds = elapsed_seconds % 60
653                      print(f"Status: FPS: {self.fps:.1f} | Faces: {len(face_locations)} | Running: {elapsed_minutes}m {remaining_seconds}s | Known faces: {len(self.known_face_names)}")
654                      last_status_update = current_time
655  
656          except KeyboardInterrupt:
657              print("\nUser interrupted the session")
658          finally:
659              self.running = False
660              self.cap.release()
661              if not self.headless:
662                  cv2.destroyAllWindows()
663  
664              # Wait for processing thread to finish
665              if self.processing_thread.is_alive():
666                  self.processing_thread.join(timeout=1)
667  
668              print(f"Session ended. Log saved to: {self.log_file}")
669              self._generate_session_summary()
670              self._generate_visualizations()
671  
672      def _generate_session_summary(self):
673          summary_file = f"{self.dirs['session']}/summary.txt"
674  
675          attendance = {}
676          with open(self.log_file, 'r') as f:
677              reader = csv.reader(f)
678              next(reader)
679              for row in reader:
680                  if len(row) >= 6:
681                      person_id = row[1]
682                      if person_id not in attendance:
683                          attendance[person_id] = {
684                              "name": row[2],
685                              "type": row[3],
686                              "count": 0,
687                              "first_seen": row[0],
688                              "last_seen": row[0],
689                              "similarities": []
690                          }
691                      attendance[person_id]["count"] += 1
692                      attendance[person_id]["last_seen"] = row[0]
693                      attendance[person_id]["similarities"].append(float(row[5]))  # Store similarity
694  
695          with open(summary_file, 'w') as f:
696              f.write(f"Session Summary: {self.session_id}\n")
697              f.write(f"Date: {datetime.now().strftime('%Y-%m-%d')}\n")
698              f.write(f"Time: {datetime.now().strftime('%H:%M:%S')}\n\n")
699  
700              f.write("Attendance Overview:\n")
701              f.write(f"Total unique individuals: {len(attendance)}\n")
702  
703              employees = {id: data for id, data in attendance.items() if data["type"] == "EMPLOYEE"}
704              visitors = {id: data for id, data in attendance.items() if data["type"] == "VISITOR"}
705  
706              f.write(f"Employees present: {len(employees)}\n")
707              f.write(f"Visitors: {len(visitors)}\n\n")
708  
709              f.write("Employee Details:\n")
710              for person_id, data in employees.items():
711                  avg_similarity = sum(data['similarities']) / len(data['similarities'])
712                  f.write(f"- {data['name']}: Detected {data['count']} times, Avg Similarity: {avg_similarity:.4f}\n")
713                  f.write(f"  First seen: {data['first_seen']}, Last seen: {data['last_seen']}\n")
714  
715              f.write("\nVisitor Details:\n")
716              for person_id, data in visitors.items():
717                  avg_similarity = sum(data['similarities']) / len(data['similarities'])
718                  f.write(f"- {person_id} (ID {person_id.split('_')[1]}): Detected {data['count']} times, Avg Similarity: {avg_similarity:.4f}\n")
719                  f.write(f"  First seen: {data['first_seen']}, Last seen: {data['last_seen']}\n")
720  
721          print(f"Session summary saved to: {summary_file}")
722          shutil.copy(self.log_file, f"{self.dirs['session']}/attendance.csv")

892      while True:
893          success, frame = system.cap.read()
894          if not success:
895              break
896  
897          frame = cv2.flip(frame, 1)
898          face_locations = system._detect_faces(frame)
899  
900          recognized_info = []
901          current_time = time.time()
902          processed_positions = set()
903  
904          for face_location in face_locations:
905              x1, y1, x2, y2 = face_location
906              center_x, center_y = (x1+x2)//2, (y1+y2)//2
907              face_pos_id = f"{center_x//20}_{center_y//20}"
908  
909              if face_pos_id in processed_positions:
910                  continue
911              processed_positions.add(face_pos_id)
912  
913              if system.auto_capture:
914                  result = system.process_new_face(frame, face_location)
915                  if result:
916                      person_id, name, bbox, similarity = result
917                      recognized_info.append((person_id, name, bbox, similarity))
918  
919          system.recognized_faces = recognized_info
920          annotated_frame = system.draw_info_on_frame(frame, face_locations)
921  
922          ret, buffer = cv2.imencode('.jpg', annotated_frame)
923          frame = buffer.tobytes()
924          yield (b'--frame\r\n'
925                 b'Content-Type: image/jpeg\r\n\r\n' + frame + b'\r\n')
